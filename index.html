ALURAFLIX do duzin import pygame # Inicializa o Pygame pygame.init() # Define as cores branco = (255, 255, 255) preto = (0, 0, 0) # Define as dimensões da tela largura = 800 altura = 600 tela = pygame.display.set_mode((largura, altura)) pygame.display.set_caption("Meu Jogo de Skate") # Cria o skate (um simples retângulo por enquanto) skate_x = largura // 2 skate_y = altura - 50 skate_largura = 50 skate_altura = 10 skate = pygame.Rect(skate_x, skate_y, skate_largura, skate_altura) # Loop principal do jogo rodando = True while rodando: for evento in pygame.event.get(): if evento.type == pygame.QUIT: rodando = False # Atualiza a posição do skate (implementar o movimento aqui) # Desenha na tela tela.fill(branco) pygame.draw.rect(tela, preto, skate) pygame.display.update() # Finaliza o Pygame pygame.quit()
